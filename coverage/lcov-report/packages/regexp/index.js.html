/**
 * 正则验证
 */
class RegExpUtils {
  /**
   * 验证内容是否符合正则
   * @param {String} value 需要验证的内容
   * @param {String} type 用来验证的正则类型，应符合regs中定义的类型，如未找到相应类型，会直接当做正则使用
   * @param  {...any} args 当正则类型定义时是使用的函数，传入的参数
   */
  test(value, type, ...args) {
    if (!value || typeof value !== 'string' || !type) return false
    let regContent = this.regs[type]
    if (!regContent) regContent = type
    // 判断是否函数形式
    if (typeof regContent === 'function') {
      regContent = regContent(...args)
    }
    return new RegExp(regContent).test(value)
  }
}
